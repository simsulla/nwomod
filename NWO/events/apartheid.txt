#setup apartheid
country_event = {
	id = 100000
	title = "EVTNAME100000"
	desc = "EVTDESC100000"
	#picture = "ship_attacked" #I'll do this later
	
	fire_only_once = yes
	
	trigger = {
		tag = SAF
		ruling_party_ideology = populist
		NOT = { has_country_modifier = apartheid }
		NOT = { has_country_flag = end_of_apartheid }
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = "EVTOPTA100000"
		add_country_modifier = {
			name = apartheid
			duration = -1
		}
		any_pop = {
			limit = {
				is_primary_culture = no
				is_accepted_culture = no
			}
			militancy = 3
			consciousness = 4
		}
		any_pop = {
			limit = {
				is_primary_culture = yes
				is_accepted_culture = yes
			}
			militancy = -2
			consciousness = 2
		}
	}
}
